NON-TERMINALS

spec ::= ( <FORMALIZE> <AS> language <SEMICOLON> )? model
language ::= <PROMELA>
| <SMV>
| <ALLOY>
| <VHDL>
model ::= ( <MODEL> ( <ID> )? <LCURLY> modelbody <RCURLY> )
modelbody ::= ( modelstmt )+
modelstmt ::= ( <DRIVERFILE> <ID> <SEMICOLON> )
| ( <CLASS> <ID> <LCURLY> ( classbody )? <RCURLY> )
| ( <ASSOCIATION> <ID> <LCURLY> assocbody <RCURLY> )
| ( <AGGREGATION> <ID> <LCURLY> agbody <RCURLY> )
| ( <ENUM> <ID> <LPARENS> <ID> ( <COMMA> <ID> )* <RPARENS> <SEMICOLON> )
classbody ::= ( classstmt )+
classstmt ::= signal
| cstate
| ccstate
| init
| join
| state
| instvar
| ( <AGSUBCONST> <ADORNS> <ID> <LCURLY> agsubbody <RCURLY> )
| ( <GENERALIZATION> <LCURLY> genbody <RCURLY> )
| ordconst
signal ::= <SIGNAL> <ID> ( <LPARENS> ( <ID> )? <RPARENS> )? <SEMICOLON>
cstate ::= <CSTATE> <ID> <LCURLY> ( cstatebody )? <RCURLY>
cstatebody ::= ( cstatestmt )+
cstatestmt ::= state
| init
| cstate
| join
| history
| ccstate
| transition
| actofstate
state ::= <STATE> <ID> <LCURLY> ( statebody )? <RCURLY>
statebody ::= ( statestmt )+
statestmt ::= transition
| actofstate
transition ::= <TRANS> ( <DQUOTE> ( transitionbody )? <ENDQUOTE> )? <TO> <ID> <SEMICOLON>
actofstate ::= ( <ACTION> <DQUOTE> ( transitionbody )? <ENDQUOTE> <SEMICOLON> )
| ( <INVARIANT> <DQUOTE> <INVTYPENAME> <SLASH> <INVARIANTEXPRESSION> <ENDQUOTE> <SEMICOLON> )
ccstate ::= <CCSTATE> <ID> <LCURLY> ccstatebody <RCURLY>
ccstatebody ::= ( ccstatestmt )+
ccstatestmt ::= cstate
init ::= <INIT> ( <DQUOTE> ( transitionbody )? <ENDQUOTE> )? <ID> <SEMICOLON>
join ::= <JOIN> <ID> <FROM> <ID> <TO> <ID> <SEMICOLON>
history ::= <HISTORY> ( <DQUOTE> ( transitionbody )? <ENDQUOTE> )? <ID> <SEMICOLON>
instvar ::= <INSTVAR> <ID> <ID> ( <ASSIGNOP> numid )? <SEMICOLON>
assocbody ::= <ID> <LBRACKET> <ID> <RBRACKET> <ID> <LBRACKET> <ID> <RBRACKET> ( ordconst )?
agsubbody ::= <ID>
genbody ::= <SUPER> <ID> <SUB> <ID>
ordconst ::= <ORDCONST> <ADORNS> <ID>
agbody ::= <WHOLE> <ID> <LBRACKET> <ID> <RBRACKET> <PART> <ID> <LBRACKET> <ID> <RBRACKET>
transitionbody ::= ( ( event ( guard )? ( <SLASH> actions )? ( <CARET> messages )? ) | ( guard ( <SLASH> actions )? ( <CARET> messages )? ) | ( <SLASH> actions ( <CARET> messages )? ) | ( <CARET> messages ) )
event ::= ( ( <ID> | <FULLID> ) ( <LPARENS> ( expression )? <RPARENS> )? )
| ( <WHEN> <LPARENS> ( expression | <ID> ) <RPARENS> )
guard ::= <EXPRESSION>
actions ::= action ( <SEMICOLON> action | <SEMICOLON> )*
action ::= ( <NEW> <LPARENS> <ID> <RPARENS> )
| ( <SEND> <LPARENS> message <RPARENS> )
| printstmt
| assignstmt
| functionInAction
messages ::= message ( <CARET> message )*
message ::= ( <ID> | <FULLID> ) ( <LPARENS> ( expression )? <RPARENS> )?
assignstmt ::= ( <ID> | <FULLID> ) <ASSIGNOP> expression
printstmt ::= <PRINT> <LPARENS> <PRINTEXPRESSION> ( <COMMA> parameterList )? <RPARENS>
functionInAction ::= ( ( <ID> | <FULLID> ) <LPARENS> ( parameterList )? <RPARENS> )
function ::= ( ( <ID> | <FULLID> ) <LPARENS> ( parameterList )? <RPARENS> )
functioninassign ::= ( <ID> | <FULLID> ) <LPARENS> parameterList <RPARENS>
parameterList ::= parm ( <COMMA> parm )*
parm ::= expression
pred ::= ( numid <COMPARE_OP> numid )
| ( <IN> <LPARENS> <ID> <RPARENS> )
expression ::= logicalExpression
logicalExpression ::= booleanExpression ( <OR> booleanExpression )*
booleanExpression ::= equalityExpression ( <AND> equalityExpression )*
equalityExpression ::= relationalExpression ( <EQUALITY_OP> relationalExpression )*
relationalExpression ::= additiveExpression ( <COMPARE_OP> additiveExpression )*
additiveExpression ::= multiplicativeExpression ( ( <PLUS> | <MINUS> ) multiplicativeExpression )*
multiplicativeExpression ::= unaryExpression ( ( <TIMES> | <SLASH> ) unaryExpression )*
unaryExpression ::= ( primaryExpression | <NOT> primaryExpression | <MINUS> primaryExpression )
primaryExpression ::= ( <LPARENS> logicalExpression <RPARENS> | function | numid )
numid ::= ( <ID> | <FULLID> | <NUM> )
